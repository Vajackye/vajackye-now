### 高精度模板
```c++
#include<bits/stdc++.h>
using namespace std;
int LEN = **;
int a[LEN];
int b[LEN];
int c[LEN];
//字符串清零
void clear(int a[])
{
  for(int i = 0; i < LEN/*字符串长度*/; i++)
    a[i] = 0;
}

//读入字符串，反向存储在a数组内，通常都是反向存储，便于字符串移动
void read(int a[])
{
  char s[LEN];
  cin>>s;//scanf("%s", s);
  clear(a);//存储运算结果,先归零
  int len = strlen(s);
  for(int i = 0; i < len; i++)
    a[len-i-1] = s[i] - '0'; //反向存储
}

//打印字符串，反向输出为原字符串，并且清除前导0
void print(int a[])
{
  int i;
  for(i = LEN - 1; i >= 1; i--)//不是i>=0是因为当整个数字等于0时仍希望输出一个字符0，即结果为0，也要输出0结果
  {
    if(a[i]!=0)//反向存储，开始清除前导0
      break;
  for(; i >=0; i--)
    putchar(a[i] + '0')//输出
  putchar('\n');
}

//加法处理
void add(int a[], int b[], int c[])
{
  clear(c);
  for(int i = 0; i < LEN - 1; i++)
  {
    c[i] += a[i] + b[i];//a,b已经在read处理过了
    if (c[i) >= 10)
    {
      c[i+1] += 1;
      c[i] -= 10;
    }
  }
}

//减法处理, 若 a < b, 此代码不能处理，要反过来写
void sub(int a[], int b[], int c[])
{
  clear(c);
  for(int i = 0; i < LEN - 1; i++)
  {
    c[i] += a[i] - b[i];
    if(c[i] < 0)
    {
      c[i+1] -= 1;
      c[i] += 10;
    }
  }
}
```
